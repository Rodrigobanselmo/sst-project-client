# SRadioCheckbox

Selectors allow users to take actions, select multiple or a single options. Supports all default HTML Text, Input and Div properties.

## Import

```tsx
import { SRadioCheckbox } from '@camel/camel-react';
```

## Usage

```tsx
<SRadioCheckbox options={['2020', '2021', '2022', '2023']} columns={2} />
```

## Structure

```tsx
const SRadioCheckbox: FC<SRadioCheckboxType> = ({ inputProps, itemProps, gridItemsProps, ...props }) => {
  return (
    <Grid container {...props}>
      {options.map((option) => {
        return (
         <GridItem {...gridItemsProps} >
            <input {...inputProps} />
            <Item {...itemProps} >
              {value || content}
            </Item>
          </GridItem>
        );
      })}
    </Grid>
```
